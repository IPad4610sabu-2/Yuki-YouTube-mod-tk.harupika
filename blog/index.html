// Key document.addEventListener('DOMContentLoaded', function() { const eventKey = JSON.parse(localStorage.getItem('eventKey')) || ['Ctrl', 'E'] const pLink = localStorage.getItem('pLink') || 'https://classroom.google.com/' let pressedKeys = [] document.addEventListener('keydown', function(event) { pressedKeys.push(event.key) if (pressedKeys.length > eventKey.length) { pressedKeys.shift() } if (eventKey.every((key, index) => key === pressedKeys[index])) { window.location.href = pLink pressedKeys = [] } }) })
